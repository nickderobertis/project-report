def _created_then_name_key(data: dict) -> str:
    created: Optional[datetime.datetime] = data['created']
    if created is None:
        created_str = '0000-00-00 00:00:00.000000'
    else:
        created_str = created.strftime('%Y-%m-%d %H:%M:%S.%f')

    name: str = data['name']
    return f'{created_str}_{name}'
